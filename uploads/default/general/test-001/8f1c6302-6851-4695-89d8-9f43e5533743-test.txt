#!/bin/bash
TOKEN=$(curl -s -X POST http://localhost:5400/api/v1/auth/login -H "Content-Type: application/json" -d '{"email":"test-modules@test.com","password":"TestPass1234!@#$"}' | grep -o '"accessToken":"[^"]*"' | cut -d'"' -f4)
BASE="http://localhost:5400/api/v1"
AUTH="Authorization: Bearer $TOKEN"
CT="Content-Type: application/json"

PASS=0
FAIL=0
RESULTS=""

test_ep() {
  local method=$1 path=$2 data=$3 label=$4
  if [ "$method" = "GET" ]; then
    resp=$(curl -s -w "\n%{http_code}" -X GET "$BASE$path" -H "$AUTH" -H "$CT" 2>&1)
  else
    resp=$(curl -s -w "\n%{http_code}" -X POST "$BASE$path" -H "$AUTH" -H "$CT" -d "$data" 2>&1)
  fi
  code=$(echo "$resp" | tail -1)
  body=$(echo "$resp" | sed '$d')
  if [ "$code" -ge 200 ] 2>/dev/null && [ "$code" -lt 300 ] 2>/dev/null; then
    line="PASS | $method $path | $code | $label"
    echo "$line"
    RESULTS="$RESULTS\n$line"
    PASS=$((PASS+1))
    # Extract ID if present
    echo "$body" | grep -o '"id":"[^"]*"' | head -1 | cut -d'"' -f4
  else
    error=$(echo "$body" | grep -o '"detail":"[^"]*"' | head -1 | cut -d'"' -f4)
    [ -z "$error" ] && error=$(echo "$body" | head -c 200)
    line="FAIL | $method $path | $code | $label | $error"
    echo "$line"
    RESULTS="$RESULTS\n$line"
    FAIL=$((FAIL+1))
  fi
}
