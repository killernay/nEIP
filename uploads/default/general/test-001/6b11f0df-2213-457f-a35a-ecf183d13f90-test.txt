#!/bin/bash

BASE="http://localhost:5400/api/v1"
AUTH="Authorization: Bearer $TOKEN"
CT="Content-Type: application/json"

PASS=0; FAIL=0; TOTAL=0

t() {
  local method=$1 path=$2 data=$3 label=$4
  TOTAL=$((TOTAL+1))
  if [ "$method" = "GET" ]; then
    resp=$(curl -s -w "\n%{http_code}" "$BASE$path" -H "$AUTH" -H "$CT" 2>&1)
  elif [ "$method" = "UPLOAD" ]; then
    resp=$(curl -s -w "\n%{http_code}" -X POST "$BASE$path" -H "$AUTH" -F "file=@/tmp/test_final.sh;filename=test.txt" -F "entityType=general" -F "entityId=test-001" 2>&1)
  else
    resp=$(curl -s -w "\n%{http_code}" -X POST "$BASE$path" -H "$AUTH" -H "$CT" -d "$data" 2>&1)
  fi
  code=$(echo "$resp" | tail -1)
  body=$(echo "$resp" | sed '$d')
  echo "$body" > /tmp/last_body.json
  if [ "$code" -ge 200 ] 2>/dev/null && [ "$code" -lt 300 ] 2>/dev/null; then
    echo "PASS | $method $path | $code | $label"
    PASS=$((PASS+1))
  else
    error=$(echo "$body" | grep -o '"detail":"[^"]*"' | head -1 | cut -d'"' -f4)
    [ -z "$error" ] && error=$(echo "$body" | head -c 150)
    echo "FAIL | $method $path | $code | $label | $error"
    FAIL=$((FAIL+1))
  fi
}
